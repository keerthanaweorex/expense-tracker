<script>
let data = [];
let type = "income";

function add() {
    let desc = document.getElementById("desc").value;
    let amount = Number(document.getElementById("amount").value);
    let date = document.getElementById("date").value;
    let category = document.getElementById("category").value;
    let note = document.getElementById("note").value;

    if(!desc || !amount) return;

    if(type === "expense") amount = -Math.abs(amount);

    data.push({desc, amount, date, category, note});
    render();
}

function render() {
    let list = document.getElementById("list");
    let bars = document.getElementById("bars");

    list.innerHTML = "";
    bars.innerHTML = "";

    let income = 0, expense = 0;

    let catTotal = {};

    data.forEach((d, i) => {

        let div = document.createElement("div");
        div.className = "transaction";

        div.innerHTML = `
            <div>
                <b>${d.desc}</b><br>
                ${d.category} - ${d.date}
            </div>
            <div>
                <span class="${d.amount>0?'green':'red'}">
                    ₹${d.amount}
                </span>
                <button onclick="del(${i})">X</button>
            </div>
        `;

        list.appendChild(div);

        if(d.amount > 0) income += d.amount;
        else expense += d.amount;

        if(d.amount < 0) {
            catTotal[d.category] = (catTotal[d.category] || 0) + Math.abs(d.amount);
        }
    });

    document.getElementById("income").innerText = "₹" + income;
    document.getElementById("expense").innerText = "₹" + Math.abs(expense);
    document.getElementById("balance").innerText = "₹" + (income + expense);


    for(let c in catTotal) {
        let val = catTotal[c];

        let div = document.createElement("div");
        div.innerHTML = `
            ${c} ₹${val}
            <div class="bar" style="width:${val/50}px;background:teal;"></div>
        `;
        bars.appendChild(div);
    }
}

function del(i) {
    data.splice(i,1);
    render();
}
